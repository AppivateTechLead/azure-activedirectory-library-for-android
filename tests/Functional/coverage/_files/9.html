483    /**
484     * extract AuthenticationResult object from response body if available
485     * 
486     * @param webResponse
487     * @return
488     */
489    private AuthenticationResult processTokenResponse(HttpWebResponse webResponse) {
490        AuthenticationResult result = new AuthenticationResult();
491        HashMap<String, String> responseItems = new HashMap<String, String>();
492        String correlationIdInHeader = null;
493        if (webResponse.getResponseHeaders() != null
494                && webResponse.getResponseHeaders().containsKey(
495                        AuthenticationConstants.AAD.CLIENT_REQUEST_ID)) {
496            // headers are returning as a list
497            List<String> listOfHeaders = webResponse.getResponseHeaders().get(
498                    AuthenticationConstants.AAD.CLIENT_REQUEST_ID);
499            if (listOfHeaders != null && listOfHeaders.size() > 0) {
500                correlationIdInHeader = listOfHeaders.get(0);
501            }
502        }
503 
504        if (webResponse.getBody() != null && webResponse.getBody().length > 0) {
505 
506            // invalid refresh token calls has error related items in the body.
507            // Status is 400 for those.
508            try {
509                String jsonStr = new String(webResponse.getBody());
510                extractJsonObjects(responseItems, jsonStr);
511                result = processUIResponseParams(responseItems);
512            } catch (final Exception ex) {
513                // There is no recovery possible here, so
514                // catch the
515                // generic Exception
516                Logger.e(TAG, ex.getMessage(), "", ADALError.SERVER_INVALID_JSON_RESPONSE, ex);
517                result = new AuthenticationResult(JSON_PARSING_ERROR, ex.getMessage());
518            }
519        } else {
520            String errMessage = null;
521            byte[] message = webResponse.getBody();
522            if (message != null) {
523                errMessage = new String(message);
524            } else {
525                errMessage = "Status code:" + String.valueOf(webResponse.getStatusCode());
526            }
527            Logger.v(TAG, "Server error message:" + errMessage);
528            result = new AuthenticationResult(String.valueOf(webResponse.getStatusCode()),
529                    errMessage);
530        }
531 
532        // Set correlationId in the result
533        if (correlationIdInHeader != null && !correlationIdInHeader.isEmpty()) {
534            try {
535                UUID correlation = UUID.fromString(correlationIdInHeader);
536                if (correlation != mRequest.getCorrelationId()) {
537                    Logger.w(TAG, "CorrelationId is not matching", "",
538                            ADALError.CORRELATION_ID_NOT_MATCHING_REQUEST_RESPONSE);
539                }
540 
541                Logger.v(TAG, "Response correlationId:" + correlationIdInHeader);
542            } catch (Exception ex) {
543                Logger.e(TAG, "Wrong format of the correlation ID:" + correlationIdInHeader, "",
544                        ADALError.CORRELATION_ID_FORMAT, ex);
545            }
546        }
547 
548        return result;
549    }
