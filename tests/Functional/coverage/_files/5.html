422                            + correlationInfo);

424                            new AuthenticationCancelError());

426                } else if (resultCode == AuthenticationConstants.UIResponse.BROWSER_CODE_ERROR) {
427                    String errCode = extras
428                            .getString(AuthenticationConstants.Browser.RESPONSE_ERROR_CODE);
429                    String errMessage = extras

434                            ADALError.SERVER_INVALID_REQUEST, errCode + " " + errMessage));

441                            .getString(AuthenticationConstants.Browser.RESPONSE_FINAL_URL);

445                                + authenticationRequest.getLogInfo());

447                                new IllegalArgumentException(
448                                        "Webview did not reach the redirectUrl"));

451                        // for token

453                                waitingRequest.mDelagete);
454 
455                        // Executes all the calls inside the Runnable to return
456                        // immediately to
457                        // UI thread. All UI
458                        // related actions will be performed using the Handler.
459                        sThreadExecutor.submit(new Runnable() {
460 
461                            @Override
462                            public void run() {
463                                Logger.v(
464                                        TAG,
465                                        "Processing url for token. "
466                                                + authenticationRequest.getLogInfo());

506                                    }
507                                } finally {
508                                    removeWaitingRequest(requestId);
509                                }
510                            }
511                        });
512                    }
513                }
514            }

615     * of your AuthenticationCallback.

771    private void acquireTokenAfterValidation(CallbackHandler callbackHandle,
772            final Activity activity, final AuthenticationRequest request) {
773        Logger.v(TAG, "Token request started" + getCorrelationLogInfo());
774 
775        // Lookup access token from cache
776        AuthenticationResult cachedItem = getItemFromCache(request);
777        if (request.getPrompt() != PromptBehavior.Always && isValidCache(cachedItem)) {
778            Logger.v(TAG, "Token is returned from cache" + getCorrelationLogInfo());
779            callbackHandle.onSuccess(cachedItem);
780            return;
781        }
782 
783        Logger.v(TAG, "Checking refresh tokens" + getCorrelationLogInfo());
784        RefreshItem refreshItem = getRefreshToken(request);
785 
786        if (request.getPrompt() != PromptBehavior.Always && refreshItem != null
787                && !StringExtensions.IsNullOrBlank(refreshItem.mRefreshToken)) {
